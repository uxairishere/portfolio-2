<script>
  var mySong = document.getElementById("mySong");
  var ico = document.getElementById("icon");
  ico.onclick = function(){
    if(mySong.paused){
      mySong.play();
      ico.src="images/pause.png";
    }else{
      mySong.pause();
      ico.src="images/play.png";
    }
  }
</script>
